Drag and Drop Explained

A simple guide to how drag and drop works in web browsers

How Drag and Drop Works

1Setting Up Elements

First, we need to create elements that can be dragged and areas where they can be dropped.

2Making Elements Draggable

We use the HTML attribute draggable="true" to make elements draggable with the mouse.

3Drag Events

When you start dragging an element, the browser triggers events: dragstart, drag, and dragend.

4Drop Events

When you drag over a valid drop target, events are triggered: dragenter, dragover, dragleave, and drop.

5Transferring Data

We can use the dataTransfer object to store information about what's being dragged.

JavaScript Code Explained

// This variable will remember which element is being dragged
let draggedElement = null;

// Find all elements with class "list" (our draggable items)
const draggableItems = document.querySelectorAll('.list');

// Find the two boxes where we can drop items
const leftBox = document.getElementById('left');
const rightBox = document.getElementById('right');

// Make each item draggable and set up event listeners
draggableItems.forEach(item => {
    item.addEventListener('dragstart', handleDragStart);
    item.addEventListener('dragend', handleDragEnd);
});

// Set up event listeners for the drop zones (boxes)
[leftBox, rightBox].forEach(zone => {
    zone.addEventListener('dragover', handleDragOver);
    zone.addEventListener('drop', handleDrop);
    zone.addEventListener('dragenter', handleDragEnter);
    zone.addEventListener('dragleave', handleDragLeave);
});

// This function runs when you start dragging an item
function handleDragStart(e) {
    draggedElement = this; // Remember which element is being dragged
    this.classList.add('dragging'); // Add visual style for dragging
    e.dataTransfer.effectAllowed = 'move'; // Set the cursor style
    e.dataTransfer.setData('text/html', this.outerHTML); // Store item data
}

// This function runs when you stop dragging an item
function handleDragEnd(e) {
    this.classList.remove('dragging'); // Remove visual style
    draggedElement = null; // Forget the dragged element
    
    // Remove any drag-over styles from boxes
    document.querySelectorAll('.drag-over').forEach(zone => {
        zone.classList.remove('drag-over');
    });
}

// This function runs when you drag over a potential drop zone
function handleDragOver(e) {
    e.preventDefault(); // Necessary to allow dropping
    e.dataTransfer.dropEffect = 'move'; // Set cursor to move style
}

// This function runs when you enter a drop zone
function handleDragEnter(e) {
    e.preventDefault();
    this.classList.add('drag-over'); // Add visual style to show it's a drop target
}

// This function runs when you leave a drop zone
function handleDragLeave(e) {
    // Only remove style if we're actually leaving the element (not just moving between children)
    if (!this.contains(e.relatedTarget)) {
        this.classList.remove('drag-over');
    }
}

// This function runs when you drop an item
function handleDrop(e) {
    e.preventDefault();
    this.classList.remove('drag-over'); // Remove visual style
    
    // If we're dropping in a different box than where the item came from
    if (draggedElement && this !== draggedElement.parentNode) {
        this.appendChild(draggedElement); // Move the item to the new box
        
        // Add a visual effect to show the item was dropped
        draggedElement.classList.add('dropped');
        // Remove the visual effect after 300 milliseconds
        setTimeout(() => {
            draggedElement.classList.remove('dropped');
        }, 300);
    }
}

// This function adds more items to the left box when the button is clicked
function addMoreItems() {
    const items = ['Item 5', 'Item 6', 'Item 7', 'Item 8'];
    items.forEach((text) => {
        const newItem = document.createElement('div');
        newItem.className = 'list';
        newItem.draggable = true;
        newItem.textContent = text;
        
        // Set up event listeners for the new item
        newItem.addEventListener('dragstart', handleDragStart);
        newItem.addEventListener('dragend', handleDragEnd);
        
        // Add the new item to the left box
        leftBox.appendChild(newItem);
    });
}
